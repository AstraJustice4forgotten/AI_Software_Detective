FROM ubuntu:22.04
RUN apt-get update && apt-get install -y openjdk-17-jdk wget unzip
# Download and setup Ghidra
RUN wget https://ghidra-sre.org/ghidra_10.3_PUBLIC_20230726.zip -O /tmp/ghidra.zip && \
    unzip /tmp/ghidra.zip -d /opt/ && rm /tmp/ghidra.zip
WORKDIR /opt/ghidra_10.3_PUBLIC
CMD ["bash"]

FROM node:20-alpine
RUN npm install -g snyk
WORKDIR /app
CMD ["sh"]

name: AI Software Detective Auto-Deploy

on:
  push:
    branches:
      - main

jobs:
  build-and-run:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v3

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: "3.11"

      - name: Install Python dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Set API keys
        run: |
          echo "OPENAI_API_KEY=${{ secrets.OPENAI_API_KEY }}" >> $GITHUB_ENV
          echo "WAPPALYZER_API_KEY=${{ secrets.WAPPALYZER_API_KEY }}" >> $GITHUB_ENV
          echo "SNYK_TOKEN=${{ secrets.SNYK_TOKEN }}" >> $GITHUB_ENV

      - name: Build Ghidra Docker container
        run: docker build -t ghidra-container -f Dockerfile_ghidra .

      - name: Build Snyk Docker container
        run: docker build -t snyk-container -f Dockerfile_snyk .

      - name: Launch AI Software Detective
        run: nohup python AI_Software_Detective_Backend.py &

      - name: Wait for backend startup
        run: sleep 10

      - name: Health check
        run: curl -s http://localhost:8000 || echo "Backend not responding yet"

**All-in-one AI cyber detective for websites, code repos, and binaries with AI sass summaries.**

## Features
- FastAPI endpoints for website, repo, and binary scanning
- Fingerprint DB + similarity detection
- Ghidra + Snyk Docker integration
- AI-generated plain-language sassy reports
- Scheduler for daily scans
- GitHub Actions deployment

```bash
